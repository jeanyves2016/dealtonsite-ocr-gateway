from fastapi import FastAPI, UploadFile, File

app = FastAPI(title="Dealtonsite OCR Gateway")

@app.get("/")
def home():
    return {"message": "Dealtonsite OCR Gateway running"}

@app.get("/health")
def health():
    return {"status": "ok"}

@app.post("/ocr")
async def ocr(file: UploadFile = File(...)):
    return {
        "status": "received",
        "filename": file.filename
    }
